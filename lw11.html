發射器（透明）
接收器（黑色）
黑色有收到紅外線，電阻變小，電壓變小
紅外線感應的程式碼(要套上小黑套）：
//1
int val;
void setup() 
{
   Serial.begin(9600);
}
void loop() 
{
    val=analogRead(A1);
    Serial.println(val);
    delay(1000);
}
//2
#define DirR 11
#define DirL 12
#define PwmR 9
#define PwmL 10
int val;
void setup() 
{
   Serial.begin(9600);
   pinMode(DirR,OUTPUT);
   pinMode(DirL,OUTPUT);
}

void loop() 
{
    val=analogRead(A1);
    Serial.println(val);
    delay(500);
    if(val<500)
    {
        digitalWrite(DirR,1);
        digitalWrite(DirL,0);
        analogWrite(PwmL,255);
        analogWrite(PwmR,255);
        delay(3000);  //順時針轉三秒
     }
     else if(val>=500&&val<=800)
     {          
         digitalWrite(DirR,1);
         digitalWrite(DirL,1);
         analogWrite(PwmL,0);
         analogWrite(PwmR,0);
         delay(3000);  //停三秒
     }
     else
     {
          digitalWrite(DirR,1);
          digitalWrite(DirL,0);
          analogWrite(PwmL,255);
          analogWrite(PwmR,255);
          delay(3000);  //逆時針轉三秒
     }
}

#define DirR 11
#define DirL 12
#define PwmR 9
#define PwmL 10
int val;
void setup() 
{
   Serial.begin(9600);
   pinMode(DirR,OUTPUT);
   pinMode(DirL,OUTPUT);
}

void loop() 
{
    val=analogRead(A1);
    Serial.println(val);
    delay(200);
    if(val<500)
    {          
        digitalWrite(DirR,1);
        digitalWrite(DirL,1);
        analogWrite(PwmL,0);
        analogWrite(PwmR,0);
        delay(10000);  //停十秒
    }
    else
    {
          digitalWrite(DirR,1);
          digitalWrite(DirL,1);
          analogWrite(PwmL,100);
          analogWrite(PwmR,100);
          delay(5000);  //直線前進五秒
    }

}
