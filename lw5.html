控制LED亮度變化的程式碼：

      #define VRxL A3
      int xL;

      void setup( )
      {
          Serial.begin(9600);  //開啟serial port,通訊速率為9600bps
      }

      void loop( )    //讀取光敏電阻，並輸出到serial port
      {
          xL=analogRead(VRxL);
          Serial.println(xL);
          delay(100);
      }

      #define VRxL A3;
      int xL;

      void setup( )
      {
      }

      void loop( )
      {
         xL=analogRead(VRxL);
         xL=map(xL,0,1023,0,225);  //map是限制範圍
         analogWrite(6,xL);
          delay(100);
      }
